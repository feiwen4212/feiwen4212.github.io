```python
class memory_replay():
    def __init__(self, max_size=10000):
        assert max_size >= 10
        self.max_size = max_size
        self.queue = []
        self.idx = 0
    
    def __len__(self):
        return len(self.queue)
    
    def sample(self, batch_size):
        idx = np.random.choice(len(self.queue),batch_size,False)
        return [np.vstack(i) for i in zip(*[self.queue[i] for i in idx])]

    def add(self, obs, act, rew, obs_p, done):
        item = (np.float32(obs), act, np.float32(rew), np.float32(obs_p), np.float32(done))
        if len(self.queue) < self.max_size:
            self.queue.append(item)
        else:
            self.queue[self.idx] = item
        self.idx = (self.idx + 1) % self.max_size


```

<li>off-policy Learning</li>

```python
@tf.function
def train(obses, acts, rews, obses_, dones, gamma=0.99):
    dones = 1 - dones
    with tf.GradientTape() as tape:
        q_theta_cands = q_theta(obses)
        q_theta_selected = tf.reduce_sum(q_theta_cands *tf.squeeze( tf.one_hot(acts, n_act), axis=1), axis=1, keepdims=True)
        q_theta_prev_cands = q_theta_prev(obses_)
        if double_q:
            max_action = tf.argmax(q_theta(obses_))
            q_theta_selected = tf.reduce_sum(q_theta_prev_cands*max_action, axis=1, keepdims=True)
        else:
            q_theta_prev_selected= tf.reduce_max(q_theta_prev_cands, axis=1, keepdims=True)
        loss = tf.reduce_mean(tf.square(rews + gamma * q_theta_prev_selected * dones - q_theta_selected))
    grads = tape.gradient(loss, q_theta.trainable_variables)
    grads_and_vars = zip(grads, q_theta.trainable_variables)
    optimizer.apply_gradients(grads_and_vars)
    return loss
```

<h2 id="dueling-q-learning">Dueling Q-learning</h2>

```python
# sum tree
class Sum_Tree():
    def __init__(self, max_size):
        self.max_size = max_size
        capacity = 1
        while capacity < self.max_size:
            capacity *= 2
        
        self.capacity = capacity
        self.values = [0] * (capacity * 2 - 1)

    def __setitem__(self, idx, val):
        assert 0 <= idx < self.max_size
        leaf_idx = self.capacity + idx -1
        self.values[leaf_idx] = val
        p_idx = (leaf_idx - 1) // 2
        while True:
            self.values[p_idx] = self.values[p_idx*2 + 1] + self.values[p_idx*2 + 2]
            if p_idx == 0:
                break
            p_idx = (p_idx - 1)// 2
    
    def __getitem__(self, idx):
        assert 0 <= idx < self.max_size
        return self.values[self.capacity + idx -1]
    
    def get_probs(self):
        return [i/self.values[0] for i in self.values[self.capacity - 1:self.capacity + self.max_size - 1]]
```

```python
# Proportional PER
class PER_Prop(memory_replay):
    def __init__(self, max_size, alpha):
        super(PER, self).__init__(max_size)
        self.tree = Sum_Tree(self.max_size)
        self.alpha = alpha
        self.default_priority = 1.

    def add(self, *args, **kwargs):
        self.tree[self.idx] = self.default_priority
        super().add(*args, **kwargs)
    
    def sample(self, batch_size, beta):
        prob = self.tree.get_probs()
        idx = np.random.choice(len(self.queue),batch_size,False, prob)
        batch = [np.vstack(i) for i in zip(*[self.queue[i] for i in idx])]
        weights = [(self.max_size * p)**-beta for p in prob]
        max_w = max(weights)
        weights = [w/max_w for w in weights]
        return batch, weights

    def update_priorities(self, idxes, vals):
        for idx, val in zip(idxes, vals):
            self.tree[idx] = val ** self.alpha + 1e-07
```
